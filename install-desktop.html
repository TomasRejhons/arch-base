#!/bin/bash function installYay { git clone https://aur.archlinux.org/yay.git cd yay makepkg -si cd .. rm -rf yay } function installGnome { sudo pacman -Syu gnome \ gnome-extra } function installAwesome { sudo pacman -Syu awesome \ rofi \ compton \ i3lock-fancy \ xclip yay -S awesome-gnome git clone https://github.com/TomasRejhons/material-awesome.git ~/.config/awesome } function setupLightDM { sudo pacman -Syu lightdm \ lightdm-webkit2-greeter \ lightdm-webkit-theme-litarvan \ mugshot sudo systemctl disable gdm sudo systemctl enable lightdm yay -S lightdm-webkit-theme-aether } installYay installGnome installAwesome setupLightDM
